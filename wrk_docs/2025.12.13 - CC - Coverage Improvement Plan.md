```rust
#[cfg(test)]
mod g711_extended_tests {
    use super::*;

    #[test]
    fn test_ulaw_boundary_values() {
        // Test 0, 32767, -32768
        assert_eq!(decode_ulaw(encode_ulaw(0)), 0);
        assert_eq!(decode_ulaw(encode_ulaw(32767)), 32767);
        assert_eq!(decode_ulaw(encode_ulaw(-32768)), -32768);
    }

    #[test]
    fn test_alaw_roundtrip_full_range() {
        for i in -32768i16..=32767 {
            let encoded = encode_alaw(i);
            let decoded = decode_alaw(encoded);
            // Allow quantization error
            assert!((decoded - i).abs() < 100);
        }
    }
}
```

```rust
mod agent_extended_tests {

    #[tokio::test]
    async fn test_ice_state_transitions() {
        let mut agent = IceAgent::new(IceConfig::default()).unwrap();
        assert_eq!(agent.state(), IceState::New);

        agent.gather_candidates().await.unwrap();
        assert_eq!(agent.state(), IceState::Gathering);

        // Add remote candidates
        agent.add_remote_candidate(make_test_candidate()).unwrap();
        agent.start_connectivity_checks().await;
        assert_eq!(agent.state(), IceState::Checking);

        // Simulate successful check
        agent.on_binding_response(&make_success_response()).await;
        assert_eq!(agent.state(), IceState::Connected);

        let mut agent1 = IceAgent::new(IceConfig {
            role: IceRole::Controlling,
            ..Default::default()
```

### Stage 3 Implementation (Transport Mocks)

#### File: crates/mdsiprtp-transport/src/tests/tls_tests.rs

```rust
#[cfg(test)]
mod tls_mock_tests {
    use super::*;

    struct MockTlsStream {
        read_data: Vec<u8>,
        write_data: Vec<u8>,
        handshake_complete: bool,
    }

    #[tokio::test]
    async fn test_tls_handshake_success() {
        let mock_stream = MockTlsStream::new_successful();
        let tls = TlsTransport::from_mock(mock_stream);

        assert!(tls.handshake().await.is_ok());
        assert!(tls.is_connected());
    }

    #[tokio::test]
    async fn test_tls_certificate_error() {
        let mock_stream = MockTlsStream::new_cert_error();
        let tls = TlsTransport::from_mock(mock_stream);

        let result = tls.handshake().await;
        assert!(matches!(result, Err(TlsError::CertificateError(_))));
    }

    #[tokio::test]
    async fn test_tls_connection_timeout() {
        let mock_stream = MockTlsStream::new_timeout();
        let tls = TlsTransport::from_mock(mock_stream);

        let result = tokio::time::timeout(
            Duration::from_millis(100),
            tls.handshake()
        ).await;

        assert!(result.is_err()); // Timeout
    }
}
```
