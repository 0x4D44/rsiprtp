```rust
impl IceAgent {
    /// Perform RFC 8445 connectivity checks
    pub async fn start_checks(&mut self) -> Result<(), IceError> {
        // 1. Form check list from candidate pairs
        let mut check_list = self.form_check_list();

        // 2. Sort by priority (G + D formula from RFC 8445)
        check_list.sort_by(|a, b| b.priority.cmp(&a.priority));

        // 3. Perform checks with pacing (Ta = 50ms default)
        for pair in &mut check_list {
            self.perform_check(pair).await?;
            tokio::time::sleep(Duration::from_millis(50)).await;
        }

        // 4. Handle triggered checks from incoming STUN
        // 5. Nominate best valid pair
        // 6. Send keepalives
    }

    async fn perform_check(&mut self, pair: &mut CandidatePair) -> Result<(), IceError> {
        // Build STUN Binding Request with:
        // - USERNAME = remote_ufrag:local_ufrag
        // - MESSAGE-INTEGRITY using remote password
        // - ICE-CONTROLLING or ICE-CONTROLLED
        // - PRIORITY attribute
        // - USE-CANDIDATE (if nominating)

        let request = StunMessage::binding_request()
            .username(&format!("{}:{}", self.remote_ufrag, self.local_ufrag))
            .priority(pair.local.priority)
            .ice_controlling(self.tie_breaker)
            .build_with_integrity(&self.remote_password)?;

        // Send and await response with retransmission
        match self.send_stun_check(pair, request).await {
            Ok(response) => {
                pair.state = PairState::Succeeded;
                // Check for peer-reflexive candidate discovery
                if let Some(mapped) = response.xor_mapped_address() {
                    self.maybe_add_peer_reflexive(mapped, pair);
                }
            }
            Err(_) => {
                pair.state = PairState::Failed;
            }
        }
        Ok(())
    }

    /// Handle incoming STUN Binding Request (triggered check)
    pub fn handle_stun_request(&mut self, request: &StunMessage, from: SocketAddr)
        -> Result<StunMessage, IceError>;
}
```

use sha1::Sha1;
use md5::{Md5, Digest};

impl StunMessage {
    /// Add MESSAGE-INTEGRITY attribute using short-term credentials
    pub fn add_message_integrity(&mut self, password: &str) -> Result<(), StunError> {
        // Compute HMAC-SHA1 over message up to (but not including) MESSAGE-INTEGRITY
        let key = password.as_bytes();
        let mut mac = Hmac::<Sha1>::new_from_slice(key)?;
        mac.update(&self.to_bytes_without_integrity());
        let result = mac.finalize().into_bytes();

        self.attributes.push(StunAttribute::MessageIntegrity(result.to_vec()));
        Ok(())
    }

    /// Add MESSAGE-INTEGRITY using long-term credentials (TURN)
    pub fn add_message_integrity_long_term(
        &mut self,
        username: &str,
        realm: &str,
        password: &str
    ) -> Result<(), StunError> {
        // Key = MD5(username:realm:password)
        let credential = format!("{}:{}:{}", username, realm, password);
        let key = Md5::digest(credential.as_bytes());

        let mut mac = Hmac::<Sha1>::new_from_slice(&key)?;
        mac.update(&self.to_bytes_without_integrity());
        let result = mac.finalize().into_bytes();

        self.attributes.push(StunAttribute::MessageIntegrity(result.to_vec()));
        Ok(())
